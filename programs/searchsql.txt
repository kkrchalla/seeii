
SELECT  T1.URL, 
	T1.SIZE, 
	T1.LAST_UPDATED_DATE, 
	T2.TITLE, 
	T2.DESC
  FROM  URL_TAB T1,
	TITLE_DESC_TAB T2,
	(SELECT B.URL_ID AS URL_ID, COUNT(*), SUM(B.RANK) AS RANK
	   FROM WORD_TAB A,
		WORD_RANK_TAB B,
	  WHERE A.WORD IN ("?KEYWORDS?")
	    AND A.WORD_ID = B.WORD_ID
	 GROUP BY URL_ID
	 ORDER BY 2 DESC, 3 DESC
	 LIMIT ?STARTPOS? 10) AS T3
 WHERE T1.URL_ID = T2.URL_ID
   AND T1.URL_ID = T3.URL_ID
